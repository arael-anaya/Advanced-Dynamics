\section*{Problem 1}

\begin{figure}[H]
    \centering
    \includegraphics[width=0.7\linewidth]{Figures/Problem Set/Problem 1.png}
\end{figure}

\subsection*{Setup}

Let $x$ be the cart's position and $\theta$ the pendulum angle, measured counterclockwise from straight down, with $\hat n_1$ horizontal and $\hat n_2$ vertical (up). The pivot $A$ coincides with the cart's position, so the bob $B$ is at
\[
    \vec r_B = (x+l\sin\theta)\,\hat n_1 - l\cos\theta\,\hat n_2,
    \qquad
    \dot{\vec r}_B = (\dot x+l\cos\theta\,\dot\theta)\,\hat n_1 + l\sin\theta\,\dot\theta\,\hat n_2
\]

\subsection*{Part (a): Equations of Motion}

\[
    T = \tfrac12 M\dot x^2 + \tfrac12 m\big(\dot{\vec r}_B\cdot\dot{\vec r}_B\big)
      = \tfrac12(M+m)\dot x^2 + ml\cos\theta\,\dot x\dot\theta + \tfrac12 ml^2\dot\theta^2,
    \qquad
    V = -mgl\cos\theta
\]
With $L=T-V$ and generalized force $Q_x=F(t)$, $Q_\theta=0$ (the cart is the only point of application),
\[
    \frac{d}{dt}\frac{\partial L}{\partial\dot x}-\frac{\partial L}{\partial x} = F(t),
    \qquad
    \frac{d}{dt}\frac{\partial L}{\partial\dot\theta}-\frac{\partial L}{\partial\theta} = 0
\]
\begin{result}
\begin{align*}
    (M+m)\ddot x + ml\cos\theta\,\ddot\theta - ml\sin\theta\,\dot\theta^2 &= F(t)\\
    ml\cos\theta\,\ddot x + ml^2\ddot\theta + mgl\sin\theta &= 0
\end{align*}
\end{result}
(the $\dot x\dot\theta$ Coriolis-type cross terms cancel identically between the two equations).

\subsection*{Part (b): Matrix Form}

With $\vec q=[x,\theta]^T$, group the equations above as $M(\vec q)\ddot{\vec q}+C(\vec q,\dot{\vec q},t)+K(\vec q,t)=0$:
\begin{result}
\[
    M(\vec q) = \begin{bmatrix} M+m & ml\cos\theta \\ ml\cos\theta & ml^2 \end{bmatrix},
    \qquad
    C(\vec q,\dot{\vec q}) = \begin{bmatrix} -ml\sin\theta\,\dot\theta^2 \\ 0 \end{bmatrix},
    \qquad
    K(\vec q,t) = \begin{bmatrix} -F(t) \\ mgl\sin\theta \end{bmatrix}
\]
\end{result}
As expected, $M(\vec q)$ is symmetric, positive-definite, and depends only on $\theta$.

\subsection*{Part (c): Normal Modes and Natural Frequencies}

With $F(t)=-kx$ and $C\to 0$ (drop the velocity-squared term, no damping), linearize about $x=\theta=0$ ($\sin\theta\approx\theta$, $\cos\theta\approx1$):
\[
    (M+m)\ddot x + ml\ddot\theta + kx = 0,
    \qquad
    ml\ddot x + ml^2\ddot\theta + mgl\theta = 0
\]
i.e.\ $M\ddot{\vec q}+K\vec q=0$ with $M=\begin{bmatrix}M+m & ml\\ml & ml^2\end{bmatrix}$, $K=\begin{bmatrix}k&0\\0&mgl\end{bmatrix}$. Assuming $\vec q = \vec Q e^{i\omega t}$, non-trivial solutions require $\det(K-\omega^2 M)=0$:
\[
    \big(k-\omega^2(M+m)\big)\big(mgl-\omega^2 ml^2\big) - \omega^4 m^2l^2 = 0
    \;\Longrightarrow\;
    Ml\,\omega^4 - \big[kl+(M+m)g\big]\omega^2 + kg = 0
\]
\begin{result}
\[
    \omega_{1,2}^2 = \frac{\big[kl+(M+m)g\big] \pm \sqrt{\big[kl+(M+m)g\big]^2 - 4Mlkg}}{2Ml}
\]
\end{result}
with mode shapes (amplitude ratios) found from the first row of $(K-\omega_i^2M)\vec Q_i=0$:
\[
    \left(\frac{\Theta}{X}\right)_i = \frac{k-\omega_i^2(M+m)}{\omega_i^2 ml}
\]
The lower frequency $\omega_1$ corresponds to the cart and pendulum swinging in phase; the higher frequency $\omega_2$ corresponds to them swinging out of phase.

\section*{Problem 2}

\begin{figure}[H]
    \centering
    \includegraphics[width=0.6\linewidth]{Figures/Problem Set/Problem 2.png}
\end{figure}

\subsection*{Setup}

Let $\theta_1$ be measured from vertical and $\theta_2$ from rod 1, with $\phi\equiv\theta_1+\theta_2$ the absolute angle of rod 2, $y$ taken positive downward. Then
\[
    \vec r_1 = l_1\sin\theta_1\,\hat n_1 + l_1\cos\theta_1\,\hat n_2,
    \qquad
    \vec r_2 = \vec r_1 + l_2\sin\phi\,\hat n_1 + l_2\cos\phi\,\hat n_2
\]
\[
    T = \tfrac12(m_1+m_2)l_1^2\dot\theta_1^2 + \tfrac12 m_2l_2^2\dot\phi^2 + m_2l_1l_2\dot\theta_1\dot\phi\cos\theta_2,
    \qquad
    V = -(m_1+m_2)gl_1\cos\theta_1 - m_2gl_2\cos\phi
\]

\subsection*{Part (a): Coupled Equations of Motion}

Applying $\dfrac{d}{dt}\dfrac{\partial L}{\partial\dot\theta_i}-\dfrac{\partial L}{\partial\theta_i}=0$ for $i=1,2$ (with $\dot\phi=\dot\theta_1+\dot\theta_2$) and simplifying:
\begin{result}
\begin{align*}
    \big[(m_1{+}m_2)l_1^2 &+ m_2l_2^2 + 2m_2l_1l_2\cos\theta_2\big]\ddot\theta_1
    + \big[m_2l_2^2+m_2l_1l_2\cos\theta_2\big]\ddot\theta_2 \\
    &- m_2l_1l_2\sin\theta_2\,\dot\theta_2(2\dot\theta_1+\dot\theta_2)
    + (m_1{+}m_2)gl_1\sin\theta_1 + m_2gl_2\sin(\theta_1{+}\theta_2) = 0
    \\[6pt]
    l_2(\ddot\theta_1+\ddot\theta_2) &+ l_1\cos\theta_2\,\ddot\theta_1 + l_1\sin\theta_2\,\dot\theta_1^2 + g\sin(\theta_1+\theta_2) = 0
\end{align*}
\end{result}
These two equations are coupled through $\theta_2$-dependent inertia terms and cannot be solved for $\ddot\theta_1,\ddot\theta_2$ independently --- they must be inverted as a $2\times2$ linear system at each instant.

\subsection*{Part (b): Numerical Integration}

Using $m_1=m_2=1\ \si{kg}$, $l_1=l_2=1\ \si{m}$, $g=9.81\ \si{m/s^2}$, and initial conditions $\theta_1(0)=1.0$, $\theta_2(0)=0.5$, $\dot\theta_1(0)=\dot\theta_2(0)=0$, the system above is inverted for $\ddot\theta_1,\ddot\theta_2$ and integrated forward in time.

\FigureListing{Figures/Graphs/Problem2_theta_vs_time.png}{0.85\linewidth}{$\theta_1$ and $\theta_2$ vs.\ time for the base initial condition, integrated over $30\ \si{s}$ ($\gtrsim$10 small-angle periods).}{fig:p2_base}

\subsection*{Part (c): Sensitivity to Initial Conditions}

Repeating the integration with $\theta_1(0) = 1.0\pm0.1$ (holding $\theta_2(0),\dot\theta_1(0),\dot\theta_2(0)$ fixed):

\FigureListing{Figures/Graphs/Problem2_sensitivity.png}{0.85\linewidth}{$\theta_1,\theta_2$ vs.\ time for $\theta_1(0)\in\{0.9,1.0,1.1\}$.}{fig:p2_sensitivity}

The three trajectories track each other closely at first, but diverge exponentially after a few oscillations until they are completely uncorrelated --- the hallmark of chaotic, sensitive dependence on initial conditions in the double pendulum. A $0.1\ \si{rad}$ ($\approx\!6^\circ$) change in $\theta_1(0)$ is enough to make the long-time motion unpredictable, even though the equations of motion are fully deterministic.

\section*{Problem 3}

\begin{figure}[H]
    \centering
    \includegraphics[width=0.55\linewidth]{Figures/Problem Set/Problem 3.png}
\end{figure}

\subsection*{Setup}

Let $\hat n_3$ point vertically upward with gravity $-g\hat n_3$, $\phi$ the angle from the downward vertical, and $\theta$ the azimuthal rotation about $\hat n_3$. Since $|\vec r|=l$ is constant,
\[
    \vec r = l\big(\sin\phi\cos\theta\,\hat n_1+\sin\phi\sin\theta\,\hat n_2-\cos\phi\,\hat n_3\big),
    \qquad
    |\dot{\vec r}|^2 = l^2\big(\dot\phi^2+\sin^2\phi\,\dot\theta^2\big)
\]
\begin{result}
\[
    L = \tfrac12 ml^2\big(\dot\phi^2+\sin^2\phi\,\dot\theta^2\big) + mgl\cos\phi
\]
\end{result}
Notice $L$ depends on $\phi,\dot\phi,\dot\theta$ but not on $\theta$ itself --- $\theta$ is ignorable.

\subsection*{Part (a): Equations of Motion}

\[
    \frac{d}{dt}\frac{\partial L}{\partial\dot\theta}-\frac{\partial L}{\partial\theta}=0,
    \qquad
    \frac{d}{dt}\frac{\partial L}{\partial\dot\phi}-\frac{\partial L}{\partial\phi}=0
\]
\begin{result}
\begin{align*}
    \frac{d}{dt}\Big(ml^2\sin^2\phi\,\dot\theta\Big) &= 0\\
    l\ddot\phi - l\sin\phi\cos\phi\,\dot\theta^2 + g\sin\phi &= 0
\end{align*}
\end{result}

\subsection*{Part (b): General Proof for an Ignorable Coordinate}

If $L=L(q_1,\dots,\dot q,\dots,t)$ does not depend on some $q$, Lagrange's equation for that coordinate is
\[
    \frac{d}{dt}\frac{\partial L}{\partial\dot q}-\underbrace{\frac{\partial L}{\partial q}}_{=\,0}=0
    \;\Longrightarrow\;
    \frac{d}{dt}\frac{\partial L}{\partial\dot q}=0
\]
\begin{result}
\[
    \frac{\partial L}{\partial\dot q} = B_q = \text{constant}
\]
\end{result}

\subsection*{Part (c): First-Order ODE for $\theta$}

\[
    B_\theta = \frac{\partial L}{\partial\dot\theta} = ml^2\sin^2\phi\,\dot\theta
\]
\begin{result}
\[
    \dot\theta = f(\phi,B_\theta) = \frac{B_\theta}{ml^2\sin^2\phi}
\]
\end{result}

\subsection*{Part (d): Routhian Reduction}

Substituting $\dot\theta=B_\theta/(ml^2\sin^2\phi)$ into $R=L-\dot\theta B_\theta$:
\[
    R(\phi,\dot\phi;B_\theta) = \tfrac12 ml^2\dot\phi^2 - \frac{B_\theta^2}{2ml^2\sin^2\phi} + mgl\cos\phi
\]
Applying Routh's equation $\dfrac{d}{dt}\Big(\dfrac{\partial R}{\partial\dot\phi}\Big)-\dfrac{\partial R}{\partial\phi}=0$:
\[
    ml^2\ddot\phi - \left(\frac{B_\theta^2\cos\phi}{ml^2\sin^3\phi} - mgl\sin\phi\right) = 0
\]
\begin{result}
\[
    \ddot\phi = \frac{B_\theta^2\cos\phi}{m^2l^4\sin^3\phi} - \frac{g}{l}\sin\phi
\]
\end{result}
which matches Part (a) directly upon substituting $\dot\theta=f(\phi,B_\theta)$ into the $\phi$ equation there --- the Routhian reduces the problem to a single second-order ODE in $\phi$ alone, with $\theta(t)$ recovered afterward from Part (c).

\section*{Problem 4}

\begin{figure}[H]
    \centering
    \includegraphics[width=0.55\linewidth]{Figures/Problem Set/Problem 4.png}
\end{figure}

\subsection*{Setup}

Let $\hat b_1=\cos\theta\,\hat n_1+\sin\theta\,\hat n_2$ (along the cart, skate-to-center-of-mass direction) and $\hat b_2=-\sin\theta\,\hat n_1+\cos\theta\,\hat n_2$. With $\vec q=(x,y,\theta)$ the skate position and cart heading,
\[
    \vec r_B = x\hat n_1+y\hat n_2,
    \qquad
    \vec r_C = \vec r_B + d\hat b_1
\]

\subsection*{Part (a): Constraint Equation}

Since $\hat n_1,\hat n_2$ are inertial, direct differentiation gives $\dot{\vec r}_B=\dot x\hat n_1+\dot y\hat n_2$; projecting onto the body frame,
\[
    \dot{\vec r}_B\cdot\hat b_1 = \dot x\cos\theta+\dot y\sin\theta,
    \qquad
    \dot{\vec r}_B\cdot\hat b_2 = -\dot x\sin\theta+\dot y\cos\theta
\]
The knife-edge constrains the skate velocity to have no component along $\hat b_2$ (no lateral slip):
\begin{result}
\[
    \phi(\vec q,\dot{\vec q}):\quad \dot y\cos\theta - \dot x\sin\theta = 0
\]
\end{result}

\subsection*{Part (b): Equations of Motion}

Using the transport theorem, $\dot{\vec r}_C = \dot{\vec r}_B + \dot\theta\,\hat n_3\times(d\hat b_1) = \dot{\vec r}_B + d\dot\theta\,\hat b_2$, so
\[
    \dot{\vec r}_C\cdot\dot{\vec r}_C = \dot x^2+\dot y^2+d^2\dot\theta^2+2d\dot\theta(\dot y\cos\theta-\dot x\sin\theta)
\]
\[
    T = \tfrac12 m(\dot x^2+\dot y^2) + \tfrac12(md^2+I_c)\dot\theta^2 + md\dot\theta(\dot y\cos\theta-\dot x\sin\theta),
    \qquad V=0
\]
With $L=T$ and the nonholonomic constraint enforced via a multiplier $\lambda$ (generalized force $\lambda\,\partial\phi/\partial\dot q_i$ conjugate to each $q_i$):
\[
    \frac{d}{dt}\frac{\partial L}{\partial\dot x}-\frac{\partial L}{\partial x}=-\lambda\sin\theta,
    \quad
    \frac{d}{dt}\frac{\partial L}{\partial\dot y}-\frac{\partial L}{\partial y}=\lambda\cos\theta,
    \quad
    \frac{d}{dt}\frac{\partial L}{\partial\dot\theta}-\frac{\partial L}{\partial\theta}=0
\]
Carrying out the derivatives (the $\dot\theta^2$ cross terms cancel in the $\theta$-equation exactly as in Problems 1--2) gives
\begin{result}
\begin{align*}
    m\ddot x - md\ddot\theta\sin\theta - md\dot\theta^2\cos\theta + \lambda\sin\theta &= 0\\
    m\ddot y + md\ddot\theta\cos\theta - md\dot\theta^2\sin\theta - \lambda\cos\theta &= 0\\
    (md^2+I_c)\ddot\theta + md(\ddot y\cos\theta-\ddot x\sin\theta) &= 0
\end{align*}
\end{result}
Introduce the skate speed $u\equiv\dot{\vec r}_B\cdot\hat b_1=\dot x\cos\theta+\dot y\sin\theta$ (so, by the constraint, $\dot x=u\cos\theta$, $\dot y=u\sin\theta$ exactly) and $\omega\equiv\dot\theta$. Substituting and simplifying the system above collapses to two clean, decoupled-from-position ODEs:
\begin{result}
\[
    \dot u = d\omega^2,
    \qquad
    \dot\omega = -\frac{mdu}{md^2+I_c}\,\omega
\]
\end{result}
together with $\dot x=u\cos\theta$, $\dot y=u\sin\theta$, $\dot\theta=\omega$. (As a check, $E=\tfrac12mu^2+\tfrac12(md^2+I_c)\omega^2$ is conserved: $\dot E = mu\dot u+(md^2+I_c)\omega\dot\omega=mdu\omega^2-mdu\omega^2=0$, consistent with the ideal, workless constraint force.)

\subsection*{Part (c): Numerical Integration}

Using $m=1\ \si{kg}$, $I_c=0.5\ \si{kg\cdot m^2}$, $d=0.5\ \si{m}$, the reduced system $(u,\omega,x,y,\theta)$ is integrated forward in time with an IVP solver for a variety of initial $(u_0,\omega_0)$.

\FigureListing{Figures/Graphs/Problem4_trajectory.png}{0.8\linewidth}{Skate path $(x(t),y(t))$ for several initial conditions.}{fig:p4_traj}

\subsection*{Part (d): Body-Frame Velocity of the Center of Mass}

From the transport-theorem relation in Part (b), $\dot{\vec r}_C = u\hat b_1+d\omega\hat b_2$, so directly
\begin{result}
\[
    v_1 = u, \qquad v_2 = d\omega
\]
\end{result}

\FigureListing{Figures/Graphs/Problem4_v1_v2.png}{0.8\linewidth}{$v_2$ vs.\ $v_1$ for several initial conditions.}{fig:p4_v1v2}

Since $\dot u=d\omega^2\geq0$ and $\omega$ decays toward zero (its sign never flips, as $\dot\omega/\omega<0$ whenever $u>0$), every trajectory in the $v_1$--$v_2$ plane moves from its initial point monotonically toward the $v_1$-axis: the cart's rotation self-damps and it asymptotically settles into straight-line motion at constant speed, a classical stability result for the knife-edge sleigh.
